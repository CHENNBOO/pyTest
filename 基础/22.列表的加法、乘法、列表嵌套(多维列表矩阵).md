# 初始化两个列表
s = [1, 2, 3]
t = [4, 5, 6]

# 列表相加：拼接两个列表
print(s + t)  # 输出: [1, 2, 3, 4, 5, 6]

# 列表乘以整数：重复列表多次
print(s * 3)  # 输出: [1, 2, 3, 1, 2, 3, 1, 2, 3]

# 初始化矩阵
matrix = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]

# 访问第一行
print(matrix[0])        # 输出: [1, 2, 3]

# 访问第一行第一个元素
print(matrix[0][0])     # 输出: 1

# 访问第二行第二个元素
print(matrix[1][1])     # 输出: 5

# 访问第三行第三个元素
print(matrix[2][2])     # 输出: 9

# 创建一个3x3的零矩阵（正确方式）
A = [0] * 3
for i in range(3):
    A[i] = [0] * 3
print(A)
# 输出: [[0, 0, 0], [0, 0, 0], [0, 0, 0]]

# 错误方式创建二维列表（所有行共享同一引用）,也就是三个列表元素，引用是同一个[0, 0, 0]
B = [[0] * 3] * 3
print(B)
# 输出: [[0, 0, 0], [0, 0, 0], [0, 0, 0]]

# 修改 A 的中间元素
A[1][1] = 1
print(A)
# 输出: [[0, 0, 0], [0, 1, 0], [0, 0, 0]]

# 修改 B 的中间元素（会同时影响所有行！）
B[1][1] = 1
print(B)
# 输出: [[0, 1, 0], [0, 1, 0], [0, 1, 0]] ← 每行都变了！

# 字符串比较
x = "FishC"
y = "FishC"
print(x is y)  # 输出: True（字符串是不可变对象，Python 会缓存相同字符串）

# 列表比较
x = [1, 2, 3]
y = [1, 2, 3]
print(x is y)  # 输出: False（两个独立列表对象，id 不同）

# 正确创建的二维列表 A（每个子列表独立）
A = [[0]*3 for _ in range(3)]
print(A[0] is A[1])  # False
print(A[1] is A[2])  # False

# 错误创建的二维列表 B（所有子列表共享同一个引用）
B = [[0]*3] * 3
print(B[0] is B[1])  # True
print(B[1] is B[2])  # True
