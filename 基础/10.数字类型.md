# =============================================
# 1. 浮点数精度问题（Float Precision Issues）
# =============================================

>>> 0.1 + 0.2
0.30000000000000004  # 不是精确的 0.3！

>>> 0.3 == 0.1 + 0.2
False  # 因为精度误差导致不相等

>>> i = 0
>>> while i < 1:
...     i = i + 0.1
...     print(i)
...
0.1
0.2
0.30000000000000004
0.4
0.5
0.6
0.7
0.7999999999999999
0.8999999999999999
0.9999999999999999
1.0999999999999999  # 超出预期，循环继续

# ❌ 问题：浮点数累加有误差，不能用于精确计数或比较！

# =============================================
# 2. 使用 decimal.Decimal 解决精度问题
# =============================================

import decimal

# 创建 Decimal 对象时用字符串表示，避免二进制转换误差
a = decimal.Decimal('0.1')
b = decimal.Decimal('0.2')
print(a + b)  # 输出: 0.3

c = decimal.Decimal('0.3')
print(a + b == c)  # 输出: True ✅ 精确相等！

# 注意：不能直接用 float 创建 Decimal
# d = decimal.Decimal(0.1)  # 这样仍然会继承 float 的精度问题！

# =============================================
# 3. 复数（Complex Numbers）
# =============================================

>>> 1 + 2j
(1+2j)

>>> x = 1 + 2j
>>> x.real  # 实部
1.0
>>> x.imag  # 虚部
2.0

# 复数运算支持基本数学操作
>>> (1 + 2j) + (3 + 4j)
(4+6j)
>>> (1 + 2j) * (3 + 4j)
(-5+10j)
