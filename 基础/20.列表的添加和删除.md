# 初始化列表
heros = ["钢铁侠", "绿巨人"]

# 使用 append 添加单个元素
heros.append("黑寡妇")
print(heros)  # 输出: ['钢铁侠', '绿巨人', '黑寡妇']

# 使用 extend 添加多个元素（可迭代对象）
heros.extend(["鹰眼", "灭霸", "雷神"])
print(heros)  # 输出: ['钢铁侠', '绿巨人', '黑寡妇', '鹰眼', '灭霸', '雷神']

# 创建数字列表
s = [1, 2, 3, 4, 5]

# 在末尾添加一个元素（使用切片赋值）
s[len(s):] = [6]
print(s)  # 输出: [1, 2, 3, 4, 5, 6]

# 再次在末尾添加多个元素
s[len(s):] = [7, 8, 9]
print(s)  # 输出: [1, 2, 3, 4, 5, 6, 7, 8, 9]

# 重新初始化 s
s = [1, 3, 4, 5]

# insert(index, value)：在指定位置插入元素
s.insert(1, 2)
print(s)  # 输出: [1, 2, 3, 4, 5]

s.insert(0, 0)
print(s)  # 输出: [0, 1, 2, 3, 4, 5]

s.insert(len(s), 6)
print(s)  # 输出: [0, 1, 2, 3, 4, 5, 6]


# 删除第一个匹配的元素 "灭霸"
heros = ['钢铁侠', '绿巨人', '黑寡妇', '鹰眼', '灭霸', '雷神']

heros.remove("灭霸")
print(heros)


# 尝试删除一个不存在的元素
heros = ['钢铁侠', '绿巨人', '黑寡妇', '鹰眼', '雷神']

try:
    heros.remove("灭霸")
except ValueError as e:
    print(f"错误：{e}")



# 使用 pop(2) 删除索引为 2 的元素，并返回该元素
heros = ['钢铁侠', '绿巨人', '黑寡妇', '鹰眼', '雷神']

removed = heros.pop(2)
print(removed)  # 输出: 黑寡妇
print(heros)    # 输出: ['钢铁侠', '绿巨人', '鹰眼', '雷神']



# 清空列表中的所有元素
heros = ['钢铁侠', '绿巨人', '黑寡妇', '鹰眼', '雷神']
heros.clear()
print(heros)  # 输出结果为[]

list.clear()：删除列表中所有元素，使列表变为空列表。
不会删除列表本身，只是清空内容。
等价于：heros[:] = [] 或 del heros[:]。
