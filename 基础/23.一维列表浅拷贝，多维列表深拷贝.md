# 一维列表的浅拷贝（copy() 和切片 [:]）
x = [1, 2, 3]
y = x.copy()  # 或 y = x[:]
x[1] = 1
print(x)  # [1, 1, 3]
print(y)  # [1, 2, 3] ← 不受影响，独立副本

# 二维列表的浅拷贝（copy()）
x = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]
y = x.copy()
x[1][1] = 0
print(x)  # [[1, 2, 3], [4, 0, 6], [7, 8, 9]]
print(y)  # [[1, 2, 3], [4, 0, 6], [7, 8, 9]] ← 内层元素被修改！

# 二维列表的深拷贝（deepcopy）
import copy
x = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]
y = copy.deepcopy(x)
x[1][1] = 0
print(x)  # [[1, 2, 3], [4, 0, 6], [7, 8, 9]]
print(y)  # [[1, 2 , 3], [4, 5, 6], [7,a, 9]] ← 完全独立，不受影响

处于效率考虑，一般使用浅拷贝，一般用切片 [:]拷贝
