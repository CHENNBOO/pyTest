# 方法1：使用 for 循环修改列表元素
oho = [1, 2, 3, 4, 5]
for i in range(len(oho)):
    oho[i] = oho[i] * 2
print(oho)  # 输出: [2, 4, 6, 8, 10]

# 方法2：使用列表推导式（推荐）
oho = [1, 2, 3, 4, 5]
oho = [i * 2 for i in oho]
print(oho)  # 输出: [2, 4, 6, 8, 10]

# 定义一个 3x3 矩阵
matrix = [[1, 2, 3],
          [4, 5, 6],
          [7, 8, 9]]

# 提取第二列（索引为1）的所有元素
col2 = [row[1] for row in matrix]
print(col2)  # 输出: [2, 5, 8]

# 提取主对角线元素（i == j 的位置）
diag = [matrix[i][i] for i in range(len(matrix))]

# 正确创建 3x3 零矩阵（每个子列表独立）
A = [0] * 3
for i in range(3):
    A[i] = [0] * 3
print(A)
# 输出: [[0, 0, 0], [0, 0, 0], [0, 0, 0]]

# 修改中间元素
A[1][1] = 1
print(A)
# 输出: [[0, 0, 0], [0, 1, 0], [0, 0, 0]]

# 使用列表推导式创建零矩阵（推荐方式）
S = [[0] * 3 for i in range(3)]
print(S)
# 输出: [[0, 0, 0], [0, 0, 0], [0, 0, 0]]

# 修改 S 的中间元素
S[1][1] = 1
print(S)
# 输出: [[0, 0, 0], [0, 1, 0], [0, 0, 0]]

# 提取 0 到 9 中的偶数（索引为偶数）

# 二维列表展平为一维列表（使用嵌套列表推导式）
matrix = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]
flatten = [col for row in matrix for col in row]
print(flatten)  # 输出: [1, 2, 3, 4, 5, 6, 7, 8, 9]

# 传统方法：使用循环和 append
flatten = []
for row in matrix:
    for col in row:
        flatten.append(col)
print(flatten)  # 输出: [1, 2, 3, 4, 5, 6, 7, 8, 9]

# 字符串组合：生成所有 x+y 的组合，笛卡尔积
result = [x + y for x in "fishc" for y in "FISHC"]
print(result)
# 输出: ['fF', 'fI', 'fS', 'fH', 'fC', 'iF', 'iI', 'iS', 'iH', 'iC', ...]

# 嵌套条件筛选：生成满足条件的坐标对，（左边被2整除，右边被3整除）笛卡尔积
pairs = [[x, y] for x in range(10) if x % 2 == 0 for y in range(10) if y % 3 == 0]
print(pairs)
# 输出: [[0, 0], [0, 3], [0, 6], [0, 9], [2, 0], [2, 3], ...]

# 传统方法：循环+条件判断
pairs = []
for x in range(10):
    if x % 2 == 0:
        for y in range(10):
            if y % 3 == 0:
                pairs.append([x, y])
print(pairs)  # 输出相同结果
