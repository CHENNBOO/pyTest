# 列表可以存不同数据类型的数据
rhyme = [1, 2, 3, 4, 5, "上山打老虎"]
print(rhyme)

for each in rhyme:
    print(each)

rhyme = [1, 2, 3, 4, 5, "上山打老虎"]

# 访问元素
print(rhyme[0])           # 输出: 1
print(rhyme[1])           # 输出: 2
print(rhyme[5])           # 输出: '上山打老虎'

# 获取列表长度
length = len(rhyme)
print(rhyme[length - 1])  # 输出: '上山打老虎'

# 使用-1索引访问最后一个元素，倒数第二个用-2
print(rhyme[-1])          # 输出: '上山打老虎'
print(rhyme[-2])          # 输出: 5

# 切片操作
rhyme = [1, 2, 3, 4, 5, "上山打老虎"]

print(rhyme[0:3]) # 从索引0到2（不包含3） [1, 2, 3]
print(rhyme[3:6]) # 从索引3到5（不包含6）[4, 5, '上山打老虎']
print(rhyme[:3]) # 从开头到索引2 [1, 2, 3]
print(rhyme[3:]) # 从索引3到结尾 [4, 5, '上山打老虎']
print(rhyme[:]) # 整个列表的副本 [1, 2, 3, 4, 5, '上山打老虎']
print(rhyme[0:6:2])  # 切片：[起始:结束:步长] 从索引0开始，到索引6（不包含），步长为2  [1, 3, 5]
print(rhyme[::2])  # 也等同于print(rhyme[0:6:2]) 
print(rhyme[::-2])  # 倒着来，从末尾开始输出，步长2
print(rhyme[::-1])  # 整个列表倒序输出
