# 初始化列表
heros = ["蜘蛛侠", "绿巨人", "黑寡妇", "鹰眼", "灭霸", "雷神"]

# 修改单个元素：将索引4的元素改为"钢铁侠"
heros[4] = "钢铁侠"
print(heros)
# 输出: ['蜘蛛侠', '绿巨人', '黑寡妇', '鹰眼', '钢铁侠', '雷神']

# 修改切片：将索引3开始到末尾的元素替换为["武松", "林冲", "李逵"]
heros[3:] = ["武松", "林冲", "李逵"]
print(heros)
# 输出: ['蜘蛛侠', '绿巨人', '黑寡妇', '武松', '林冲', '李逵']
Step one: 将赋值号 (=) 左边指定的内容删除
Step two: 将包含在赋值号 (=) 右边的可迭代对象中的片段插入左边被删除的位置
其实它是分上面两个步骤来实现的


# 数字列表
nums = [3, 1, 9, 6, 8, 3, 5, 3]

# 排序（升序）
nums.sort()
print(nums)
# 输出: [1, 3, 3, 3, 5, 6, 8, 9]

# 反转列表顺序
nums.reverse()
print(nums)
# 输出: [9, 8, 6, 5, 3, 3, 3, 1]

# 反转 heros 列表
heros.reverse()
print(heros)
# 输出: ['李逵', '林冲', '武松', '黑寡妇', '绿巨人', '蜘蛛侠']

# 重新初始化 nums
nums = [3, 1, 9, 6, 8, 3, 5, 3]

# 使用 sort(reverse=True) 实现降序排，等同于先调用sort再调用reverse
nums.sort(reverse=True)
print(nums)
# 输出: [9, 8, 6, 5, 3, 3, 3, 1]

下面是查询

# 当前 nums 状态（已排序为降序）
nums = [9, 8, 6, 5, 3, 3, 3, 1]

# 统计数字 3 出现的次数
print(nums.count(3))  # 输出: 3

# heros 列表状态：['李逵', '林冲', '武松', '黑寡妇', '绿巨人', '蜘蛛侠']
heros = ['李逵', '林冲', '武松', '黑寡妇', '绿巨人', '蜘蛛侠']

# 查找 "绿巨人" 的索引（第一个匹配的位置）
print(heros.index("绿巨人"))  # 输出: 4

# 使用 index() 修改该位置的元素
heros[heros.index("绿巨人")] = "神奇女侠"
print(heros)
# 输出: ['李逵', '林冲', '武松', '黑寡妇', '神奇女侠', '蜘蛛侠']

# 重新初始化 nums
nums = [3, 1, 9, 6, 8, 3, 5, 3]

# 查找第一个 3 的索引
print(nums.index(3))  # 输出: 0

# 查找从索引 1 开始，到索引 7 结束范围内第一个 3 的索引
print(nums.index(3, 1, 7))  # 输出: 5

# 复制列表方法一：使用 copy()
nums_copy1 = nums.copy()
print(nums_copy1)  # 输出: [3, 1, 9, 6, 8, 3, 5, 3]

# 复制列表方法二：使用切片 [:]
nums_copy2 = nums[:]
print(nums_copy2)  # 输出: [3, 1, 9, 6, 8, 3, 5, 3]
